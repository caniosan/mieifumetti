            </div>

    </main>


    <script>

        const categories = ["Dylan Dog", "Dylan Dog Ristampa", "Dylan Dog Book", "Dylan Dog Super Book", "Albo Gigante", "Diabolik", "Diabolik Ristampe", "Diabolik Annate", "Brendon", "Simpsons"];

        

        // Mock data - Qui andrebbe il fetch dal tuo file JSON

        const comics = [

            { serie: "Dylan Dog", titolo: "L'alba dei morti viventi", anno: 1986, posseduto: true, condizione: "Ottima", inVendita: false, prezzo: 0, note: "Prima edizione", img: "https://via.placeholder.com/200x300" },

            { serie: "Diabolik", titolo: "Il Re del Terrore", anno: 1962, posseduto: true, condizione: "Uscita da edicola", inVendita: true, prezzo: 50, note: "Ristampa", img: "https://via.placeholder.com/200x300" }

        ];


        function renderCategories() {

            const nav = document.getElementById('categories');

            categories.forEach(cat => {

                const btn = document.createElement('button');

                btn.className = "px-4 py-2 bg-slate-800 rounded-full text-sm hover:bg-sky-500 transition";

                btn.innerText = cat;

                btn.onclick = () => filterComics(cat);

                nav.appendChild(btn);

            });

        }


        function renderComics(data) {

            const grid = document.getElementById('comic-grid');

            grid.innerHTML = data.map(c => `

                <div class="comic-card bg-slate-900 rounded-xl overflow-hidden shadow-lg">

                    <img src="${c.img}" alt="${c.titolo}" class="w-full h-64 object-cover">

                    <div class="p-4">

                        <span class="text-xs font-bold uppercase text-sky-400">${c.serie}</span>

                        <h3 class="text-lg font-bold mt-1">${c.titolo}</h3>

                        <p class="text-sm text-slate-400">${c.anno} - Condizione: ${c.condizione}</p>

                        <div class="mt-4 flex justify-between items-center">

                            ${c.inVendita ? `<span class="bg-green-500 text-black text-xs font-bold px-2 py-1 rounded">€${c.prezzo}</span>` : '<span class="text-slate-500 text-xs text-italic">In collezione</span>'}

                            <button class="text-slate-300 hover:text-white">ℹ️ Info</button>

                        </div>

                    </div>

                </div>

            `).join('');

        }


        function filterComics(cat) {

            const filtered = comics.filter(c => c.serie === cat);

            renderComics(filtered);

        }


        renderCategories();

        renderComics(comics);

    </script>
